CREATE VIEW student_course_details AS
SELECT 
    s.student_id,
    s.student_name,
    s.major,
    c.course_id,
    c.course_name,
    c.credits,
    e.semester,
    e.grade
FROM students s
INNER JOIN enrollments e 
    ON s.student_id = e.student_id
INNER JOIN courses c 
    ON e.course_id = c.course_id;
